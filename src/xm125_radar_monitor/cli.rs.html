5/// Parse I2C address from string, supporting both decimal and hex formats
6fn parse_i2c_address(s: &str) -> Result<u16, String> {
7    if let Some(hex_str) = s.strip_prefix("0x") {
8        u16::from_str_radix(hex_str, 16).map_err(|_| format!("Invalid hex I2C address: {s}"))
9    } else {
10        s.parse::<u16>()
11            .map_err(|_| format!("Invalid I2C address: {s}"))
12    }
13}

424pub enum FirmwareAction {

426    Check,
427
428    /// Update firmware to match the specified detector mode
429    ///
430    /// Automatically flashes the correct firmware binary for the selected mode.
431    /// Uses stm32flash and GPIO control for safe firmware updates.

433        /// Target firmware type (distance, presence, or breathing)
434        #[arg(help = "Firmware type: distance, presence, or breathing")]
435        firmware_type: FirmwareType,
436
437        /// Force update even if firmware already matches
438        #[arg(short, long, help = "Force firmware update even if already correct")]
439        force: bool,
440
441        /// Verify firmware after update (adds delay and may timeout)
442        #[arg(long, help = "Verify firmware installation after update")]
443        verify: bool,
444    },
445
446    /// Verify firmware integrity using checksums
447    Verify {
448        /// Firmware type to verify against
449        firmware_type: Option<FirmwareType>,
450    },
451
452    /// Erase the XM125 chip completely
453    ///
454    /// This will completely erase all firmware from the XM125 module.
455    /// The module will need to be reprogrammed before it can be used again.
456    /// Use with caution - this operation cannot be undone.
457    Erase {
458        /// Confirm the erase operation (required for safety)
459        #[arg(long, help = "Confirm that you want to erase the chip (required)")]
460        confirm: bool,
461    },
462
463    /// Calculate and display firmware checksums
464    ///
465    /// Calculates MD5 checksums for firmware binary files to verify integrity
466    /// and compare different firmware versions. Useful for validation and
467    /// troubleshooting firmware-related issues.
468    Checksum {
469        /// Specific firmware type to checksum (if not specified, shows all)
470        #[arg(help = "Firmware type to checksum: distance, presence, or breathing")]
471        firmware_type: Option<FirmwareType>,
472
473        /// Show detailed information including file paths and sizes
474        #[arg(short, long, help = "Show detailed information about firmware files")]
475        verbose: bool,
476    },
477}
