5/// Parse I2C address from string, supporting both decimal and hex formats
6fn parse_i2c_address(s: &str) -> Result<u16, String> {
7    if let Some(hex_str) = s.strip_prefix("0x") {
8        u16::from_str_radix(hex_str, 16).map_err(|_| format!("Invalid hex I2C address: {s}"))
9    } else {
10        s.parse::<u16>()
11            .map_err(|_| format!("Invalid I2C address: {s}"))
12    }
13}

18        if let Some(device) = &self.i2c_device {
19            device.clone()
20        } else {
21            format!("/dev/i2c-{}", self.i2c_bus)
22        }
23    }

329impl From<FirmwareType> for firmware::FirmwareType {
330    fn from(cli_type: FirmwareType) -> Self {
331        match cli_type {
332            FirmwareType::Distance => firmware::FirmwareType::Distance,
333            FirmwareType::Presence => firmware::FirmwareType::Presence,
334            FirmwareType::Breathing => firmware::FirmwareType::Breathing,
335        }
336    }

339impl From<crate::cli::DetectorMode> for firmware::FirmwareType {
340    fn from(mode: crate::cli::DetectorMode) -> Self {
341        match mode {
342            DetectorMode::Distance => firmware::FirmwareType::Distance,
343            DetectorMode::Presence | DetectorMode::Combined => firmware::FirmwareType::Presence, // Default to presence for combined
344            DetectorMode::Breathing => firmware::FirmwareType::Breathing,
345        }
346    }
