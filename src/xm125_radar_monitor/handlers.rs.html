14/// Handle firmware-related commands
15pub async fn handle_firmware_action(
16    radar: &mut XM125Radar,
17    action: &FirmwareAction,
18    firmware_path: &str,
19) -> Result<(), RadarError> {
20    match action {
21        FirmwareAction::Check => {
22            let info = radar.get_info()?;
23            println!("üì¶ Current Firmware:");
24            println!("{info}");
25        }
26
27        FirmwareAction::Update { firmware_type, .. } => {
28            let manager =
29                firmware::FirmwareManager::new(firmware_path, "/usr/bin/xm125-control.sh", 0x52);
30            manager.update_firmware(*firmware_type).await?;
31        }
32
33        FirmwareAction::Verify { firmware_type } => {
34            if let Some(fw_type) = firmware_type {
35                info!("Would verify firmware type: {fw_type:?}");
36            } else {
37                info!("Would verify current firmware");
38            }
39        }
40
41        // These are handled earlier in the flow
42        FirmwareAction::Checksum { .. }
43        | FirmwareAction::Erase { .. }
44        | FirmwareAction::Bootloader { .. } => {
45            unreachable!("These actions should be handled before I2C initialization");
46        }
47    }
48    Ok(())
49}

51/// Handle firmware erase command
52pub async fn handle_firmware_erase_command(confirm: bool) -> Result<(), RadarError> {
53    if !confirm {
54        eprintln!("‚ùå Chip erase requires --confirm flag for safety");
55        eprintln!("   This will completely erase all firmware from the XM125 module.");
56        eprintln!("   Use: xm125-radar-monitor firmware erase --confirm");
57        return Err(RadarError::DeviceError {
58            message: "Erase operation not confirmed".to_string(),
59        });
60    }
61
62    println!("‚ö†Ô∏è  WARNING: This will completely erase the XM125 firmware!");
63    println!("üîÑ Starting chip erase...");
64
65    let output = Command::new("stm32flash")
66        .args(["-m", "/dev/i2c-2", "-a", "0x48"])
67        .output()
68        .map_err(|e| RadarError::DeviceError {
69            message: format!("Failed to execute stm32flash: {e}"),
70        })?;
71
72    if !output.status.success() {
73        return Err(RadarError::DeviceError {
74            message: format!(
75                "Chip erase failed: {}",
76                String::from_utf8_lossy(&output.stderr)
77            ),
78        });
79    }
80
81    println!("‚úÖ Chip erase completed successfully");
82    println!("   The XM125 module now needs firmware to be programmed before use.");
83    Ok(())
84}

86/// Handle firmware checksum command
87pub fn handle_firmware_checksum_command(
88    firmware_type: Option<&FirmwareType>,
89    verbose: bool,
90    firmware_path: &str,
91) -> Result<(), RadarError> {
92    if let Some(fw_type) = firmware_type {
93        let manager =
94            firmware::FirmwareManager::new(firmware_path, "/usr/bin/xm125-control.sh", 0x52);
95        let checksum = manager.calculate_binary_checksum(*fw_type)?;
96        if verbose {
97            println!(
98                "Firmware: {} ({})",
99                fw_type.display_name(),
100                fw_type.binary_filename()
101            );
102            println!("Path: {}/{}", firmware_path, fw_type.binary_filename());
103            println!("MD5: {checksum}");
104        } else {
105            println!("{}: {}", fw_type.display_name(), checksum);
106        }
107        Ok(())
108    } else {
109        for fw_type in [
110            firmware::FirmwareType::Distance,
111            firmware::FirmwareType::Presence,
112        ] {
113            let manager =
114                firmware::FirmwareManager::new(firmware_path, "/usr/bin/xm125-control.sh", 0x52);
115            match manager.calculate_binary_checksum(fw_type) {
116                Ok(checksum) => {
117                    if verbose {
118                        println!(
119                            "Firmware: {} ({})",
120                            fw_type.display_name(),
121                            fw_type.binary_filename()
122                        );
123                        println!("Path: {}/{}", firmware_path, fw_type.binary_filename());
124                        println!("MD5: {checksum}");
125                        println!();
126                    } else {
127                        println!("{}: {}", fw_type.display_name(), checksum);
128                    }
129                }
130                Err(e) => {
131                    eprintln!("‚ùå {}: {e}", fw_type.display_name());
132                }
133            }
134        }
135        Ok(())
136    }
137}

139/// Handle bootloader command
140pub async fn handle_bootloader_command(cli: &Cli, test_mode: bool) -> Result<(), RadarError> {
141    let _gpio_pins = cli.get_gpio_pins();
142    let mut gpio_controller = XM125GpioController::new();
143    gpio_controller.initialize()?;
144
145    if test_mode {
146        println!("üß™ Testing bootloader mode (will reset back to run mode)");
147        gpio_controller.reset_to_bootloader_mode()?;
148
149        // Wait a moment for the device to enter bootloader mode
150        tokio::time::sleep(tokio::time::Duration::from_millis(500)).await;
151
152        // Reset back to run mode
153        gpio_controller.reset_to_run_mode()?;
154        println!("‚úÖ Bootloader test completed");
155    } else {
156        println!("üîÑ Putting XM125 into bootloader mode...");
157        gpio_controller.reset_to_bootloader_mode()?;
158        println!("‚úÖ XM125 is now in bootloader mode (I2C address 0x48)");
159        println!("   Ready for firmware programming with stm32flash");
160        println!("   Use 'xm125-radar-monitor gpio reset-run' to return to normal mode");
161    }
162    Ok(())
163}

165/// Handle GPIO commands
166pub fn handle_gpio_command(cli: &Cli, action: &GpioAction) -> Result<(), RadarError> {
167    let _gpio_pins = cli.get_gpio_pins();
168    let mut gpio_controller = XM125GpioController::new();
169    gpio_controller.initialize()?;
170
171    match action {
172        GpioAction::Init => {
173            gpio_controller.initialize()?;
174            println!("‚úÖ GPIO pins initialized successfully");
175        }
176        GpioAction::Status => {
177            gpio_controller.show_gpio_status()?;
178        }
179        GpioAction::ResetRun => {
180            gpio_controller.initialize()?;
181            gpio_controller.reset_to_run_mode()?;
182            println!("‚úÖ XM125 reset to run mode (I2C address 0x52)");
183        }
184        GpioAction::ResetBootloader => {
185            gpio_controller.initialize()?;
186            gpio_controller.reset_to_bootloader_mode()?;
187            println!("‚úÖ XM125 reset to bootloader mode (I2C address 0x48)");
188        }
189        GpioAction::Test => {
190            gpio_controller.initialize()?;
191            gpio_controller.test_bootloader_control()?;
192        }
193    }
194    Ok(())
