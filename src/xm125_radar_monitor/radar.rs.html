301    pub async fn calibrate(&mut self) -> Result<()> {
302        info!("Starting XM125 calibration...");
303
304        // Send calibration command
305        self.send_command(CMD_APPLY_CONFIG_AND_CALIBRATE)?;
306
307        // Wait for calibration to complete
308        let start_time = Instant::now();
309        loop {
310            let status = self.get_status_raw()?;
311
312            if status & STATUS_CALIBRATION_DONE != 0 {
313                self.is_calibrated = true;
314                self.last_calibration = Some(Instant::now());
315                info!("XM125 calibration completed successfully");
316                return Ok(());
317            }
318
319            if status & STATUS_ERROR != 0 {
320                return Err(RadarError::DeviceError {
321                    message: "Calibration failed - device error".to_string(),
322                });
323            }
324
325            if start_time.elapsed() > CALIBRATION_TIMEOUT {
326                return Err(RadarError::Timeout { timeout: 2 });
327            }
328
329            tokio::time::sleep(Duration::from_millis(100)).await;
330        }
331    }

333    pub async fn measure_distance(&mut self) -> Result<DistanceMeasurement> {
334        // Auto-connect if not connected and auto-reconnect is enabled
335        if !self.is_connected && self.config.auto_reconnect {
336            info!("Auto-connecting for distance measurement...");
337            self.auto_connect().await?;
338        }
339
340        if !self.is_connected {
341            return Err(RadarError::NotConnected);
342        }
343
344        // Check if calibration is needed (every 5 minutes or if not calibrated)
345        // Using map_or instead of is_none_or due to stability requirements
346        #[allow(unknown_lints, clippy::unnecessary_map_or)]
347        if !self.is_calibrated
348            || self
349                .last_calibration
350                .map_or(true, |t| t.elapsed() > Duration::from_secs(300))
351        {
352            self.calibrate().await?;
353        }
354
355        // Send measurement command
356        self.send_command(CMD_MEASURE_DISTANCE)?;
357
358        // Wait for measurement to be ready
359        let start_time = Instant::now();
360        loop {
361            let status = self.get_status_raw()?;
362
363            if status & STATUS_MEASUREMENT_READY != 0 {
364                break;
365            }
366
367            if status & STATUS_ERROR != 0 {
368                return Err(RadarError::MeasurementFailed(
369                    "Device error during measurement".to_string(),
370                ));
371            }
372
373            if start_time.elapsed() > Duration::from_secs(2) {
374                return Err(RadarError::Timeout { timeout: 2 });
375            }
376
377            tokio::time::sleep(Duration::from_millis(10)).await;
378        }
379
380        // Read measurement result
381        self.read_distance_result()
382    }

506    /// Automatically connect with retry logic
507    pub async fn auto_connect(&mut self) -> Result<()> {
508        const MAX_RETRIES: u32 = 3;
509        const RETRY_DELAY: Duration = Duration::from_millis(500);
510
511        for attempt in 1..=MAX_RETRIES {
512            match self.connect_async().await {
513                Ok(()) => {
514                    info!("Successfully connected to XM125 on attempt {attempt}");
515                    return Ok(());
516                }
517                Err(RadarError::NotConnected) if attempt == 1 => {
518                    // First connection failure - try resetting XM125 to run mode
519                    warn!(
520                        "XM125 not detected on I2C bus, attempting hardware reset to run mode..."
521                    );
522                    if let Err(reset_err) = self.reset_xm125_to_run_mode() {
523                        warn!("Failed to reset XM125: {reset_err}");
524                    } else {
525                        info!("XM125 reset completed, retrying connection...");
526                        // Give the module time to initialize after reset
527                        tokio::time::sleep(Duration::from_millis(1000)).await;
528                    }
529                }
530                Err(e) => {
531                    if attempt < MAX_RETRIES {
532                        warn!("Connection attempt {attempt} failed: {e}. Retrying...");
533                        tokio::time::sleep(RETRY_DELAY).await;
534                    } else {
535                        return Err(e);
536                    }
537                }
538            }
539        }
540        Err(RadarError::NotConnected)
541    }

567        info!(

589
590        Ok(())

647    /// Wait for calibration to complete
648    async fn wait_for_calibration(&mut self) -> Result<()> {
649        let start_time = Instant::now();
650
651        loop {
652            let status = self.get_status_raw()?;
653
654            // Check if device is already calibrated (for presence detector, status 0x07 is sufficient)
655            if status & STATUS_SENSOR_CALIBRATE_OK != 0 {
656                self.is_calibrated = true;
657                self.last_calibration = Some(Instant::now());
658                info!("XM125 calibration completed successfully");
659                return Ok(());
660            }
661
662            // For presence detector, status 0x07 (bits 0,1,2) indicates ready state
663            if status == 0x07 {
664                self.is_calibrated = true;
665                self.last_calibration = Some(Instant::now());
666                info!("XM125 presence detector ready (status: 0x{status:02X})");
667                return Ok(());
668            }
669
670            if status & STATUS_ERROR != 0 {
671                return Err(RadarError::DeviceError {
672                    message: "Calibration failed - device error".to_string(),
673                });
674            }
675
676            if start_time.elapsed() > CALIBRATION_TIMEOUT {
677                return Err(RadarError::Timeout { timeout: 2 });
678            }
679
680            tokio::time::sleep(Duration::from_millis(100)).await;
681        }
682    }

684    /// Measure presence detection
685    pub async fn measure_presence(&mut self) -> Result<PresenceMeasurement> {
686        // Auto-connect if not connected and auto-reconnect is enabled
687        if !self.is_connected && self.config.auto_reconnect {
688            info!("Auto-connecting for presence measurement...");
689            self.auto_connect().await?;
690        }
691
692        if !self.is_connected {
693            return Err(RadarError::NotConnected);
694        }
695
696        // Ensure we're configured for presence detection
697        if !matches!(
698            self.config.detector_mode,
699            DetectorMode::Presence | DetectorMode::Combined
700        ) {
701            return Err(RadarError::InvalidParameters(
702                "Detector not configured for presence detection".to_string(),
703            ));
704        }
705
706        // Send presence detection command sequence
707        // 1. Apply configuration first
708        self.send_command(CMD_PRESENCE_APPLY_CONFIGURATION)?;
709
710        // Small delay for configuration to be applied
711        tokio::time::sleep(Duration::from_millis(10)).await;
712
713        // 2. Start measurement
714        self.send_command(CMD_PRESENCE_START)?;
715
716        // Wait for measurement
717        self.wait_for_measurement().await?;
718
719        // Read presence result registers individually (4 bytes each)
720        let presence_result_data = self.i2c.read_register(REG_PRESENCE_RESULT, 4)?;
721        let presence_distance_data = self.i2c.read_register(REG_PRESENCE_DISTANCE, 4)?;
722        let intra_score_data = self.i2c.read_register(REG_INTRA_PRESENCE_SCORE, 4)?;
723        let inter_score_data = self.i2c.read_register(REG_INTER_PRESENCE_SCORE, 4)?;
724
725        // Parse presence result (register 16/0x10)
726        let presence_result = u32::from_be_bytes([
727            presence_result_data[0],
728            presence_result_data[1],
729            presence_result_data[2],
730            presence_result_data[3],
731        ]);
732
733        let presence_detected = (presence_result & 0x01) != 0; // Bit 0: presence detected
734        #[allow(clippy::no_effect_underscore_binding)] // Reserved for future sticky detection logic
735        let _presence_sticky = (presence_result & 0x02) != 0; // Bit 1: presence sticky
736        let detector_error = (presence_result & 0x8000) != 0; // Bit 15: detector error
737
738        if detector_error {
739            return Err(RadarError::DeviceError {
740                message: "Presence detector error flag set".to_string(),
741            });
742        }
743
744        // Parse presence distance (register 17/0x11) - distance in mm
745        let distance_raw = u32::from_be_bytes([
746            presence_distance_data[0],
747            presence_distance_data[1],
748            presence_distance_data[2],
749            presence_distance_data[3],
750        ]);
751
752        // Parse intra presence score (register 18/0x12) - fast motion
753        let intra_score_raw = u32::from_be_bytes([
754            intra_score_data[0],
755            intra_score_data[1],
756            intra_score_data[2],
757            intra_score_data[3],
758        ]);
759
760        // Parse inter presence score (register 19/0x13) - slow motion
761        let inter_score_raw = u32::from_be_bytes([
762            inter_score_data[0],
763            inter_score_data[1],
764            inter_score_data[2],
765            inter_score_data[3],
766        ]);
767
768        #[allow(clippy::cast_precision_loss)]
769        let presence_distance = (distance_raw as f32) / 1000.0; // Convert mm to m
770        #[allow(clippy::cast_precision_loss)]
771        let intra_score = (intra_score_raw as f32) / 1000.0;
772        #[allow(clippy::cast_precision_loss)]
773        let inter_score = (inter_score_raw as f32) / 1000.0;
774
775        self.last_measurement = Some(Instant::now());
776
777        Ok(PresenceMeasurement {
778            presence_detected,
779            presence_distance,
780            intra_presence_score: intra_score,
781            inter_presence_score: inter_score,
782            timestamp: chrono::Utc::now(),
783        })
784    }

787    pub async fn measure_combined(&mut self) -> Result<CombinedMeasurement> {
788        // Auto-connect if not connected and auto-reconnect is enabled
789        if !self.is_connected && self.config.auto_reconnect {
790            info!("Auto-connecting for combined measurement...");
791            self.auto_connect().await?;
792        }
793
794        if !self.is_connected {
795            return Err(RadarError::NotConnected);
796        }
797
798        if self.config.detector_mode != DetectorMode::Combined {
799            return Err(RadarError::InvalidParameters(

807
808        Ok(CombinedMeasurement {
809            distance,
810            presence,
811            timestamp: chrono::Utc::now(),
812        })

855        let start_time = Instant::now();
856
857        loop {
858            let status = self.get_status_raw()?;
859
860            if status & STATUS_MEASUREMENT_READY != 0 {
861                return Ok(());
862            }
863
864            if status & STATUS_ERROR != 0 {
865                return Err(RadarError::DeviceError {
866                    message: "Device error during measurement".to_string(),
867                });
868            }
869
870            if start_time.elapsed() > MEASUREMENT_TIMEOUT {
871                return Err(RadarError::Timeout { timeout: 5 });
872            }
873
874            tokio::time::sleep(Duration::from_millis(10)).await;
875        }
