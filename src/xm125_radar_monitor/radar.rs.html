340impl BreathingAppState {
341    fn from_u32(value: u32) -> Self {
342        match value {
343            0 => Self::Init,
344            1 => Self::NoPresence,
345            2 => Self::IntraPresence,
346            3 => Self::DetermineDistance,
347            4 => Self::EstimateBreathingRate,
348            _ => Self::Init, // Default fallback
349        }
350    }
351
352    pub fn display_name(&self) -> &'static str {
353        match self {
354            Self::Init => "Initializing",
355            Self::NoPresence => "No Presence",
356            Self::IntraPresence => "Presence Detected",
357            Self::DetermineDistance => "Determining Distance",
358            Self::EstimateBreathingRate => "Estimating Breathing Rate",
359        }
360    }
361}

557    pub async fn calibrate(&mut self) -> Result<()> {
558        info!("Starting XM125 calibration with proper configuration sequence...");
559
560        // Step 1: Check initial status - verify no Busy or Error bits
561        let initial_status = self.get_status_raw()?;
562        if initial_status & STATUS_BUSY != 0 {
563            return Err(RadarError::DeviceError {
564                message: "XM125 is busy - cannot start calibration".to_string(),
565            });
566        }
567        if initial_status & STATUS_ALL_ERROR_MASK != 0 {
568            return Err(RadarError::DeviceError {
569                message: format!(
570                    "XM125 has error flags before calibration: 0x{:08X}",
571                    initial_status
572                ),
573            });
574        }
575
576        // Step 2: Write configuration to Start and End registers for distance detector
577        if matches!(
578            self.config.detector_mode,
579            DetectorMode::Distance | DetectorMode::Combined
580        ) {
581            info!(
582                "Configuring distance detection range: {:.2}m to {:.2}m",
583                self.config.start_m,
584                self.config.start_m + self.config.length_m
585            );
586
587            let start_mm = (self.config.start_m * 1000.0) as u32;
588            let end_mm = ((self.config.start_m + self.config.length_m) * 1000.0) as u32;
589
590            debug!("Writing Start register: {}mm", start_mm);
591            let start_bytes = start_mm.to_le_bytes();
592            self.i2c.write_register(REG_START_CONFIG, &start_bytes)?;
593
594            debug!("Writing End register: {}mm", end_mm);
595            let end_bytes = end_mm.to_le_bytes();
596            self.i2c.write_register(REG_END_CONFIG, &end_bytes)?;
597
598            tokio::time::sleep(Duration::from_millis(10)).await;
599        }
600
601        // Step 3: Send APPLY_CONFIG_AND_CALIBRATE command
602        info!("Sending APPLY_CONFIG_AND_CALIBRATE command...");
603        self.send_command(CMD_APPLY_CONFIG_AND_CALIBRATE)?;
604
605        // Step 4: Poll Detector Status until Busy bit is cleared
606        info!("Waiting for calibration to complete...");
607        let start_time = Instant::now();
608
609        loop {
610            tokio::time::sleep(Duration::from_millis(50)).await; // Poll every 50ms
611            let status = self.get_status_raw()?;
612
613            if status & STATUS_BUSY != 0 {
614                debug!("Calibration in progress... (status: 0x{:08X})", status);
615                if start_time.elapsed() > CALIBRATION_TIMEOUT {
616                    return Err(RadarError::Timeout { timeout: 2 });
617                }
618                continue;
619            }
620
621            // Busy bit cleared - check for errors
622            if status & STATUS_ALL_ERROR_MASK != 0 {
623                let mut error_details = Vec::new();
624                if status & STATUS_SENSOR_CALIBRATE_ERROR != 0 {
625                    error_details.push("Sensor calibration failed");
626                }
627                if status & STATUS_DETECTOR_CALIBRATE_ERROR != 0 {
628                    error_details.push("Detector calibration failed");
629                }
630                if status & STATUS_CONFIG_CREATE_ERROR != 0 {
631                    error_details.push("Configuration creation failed");
632                }
633                if status & STATUS_SENSOR_CREATE_ERROR != 0 {
634                    error_details.push("Sensor creation failed");
635                }
636                if status & STATUS_DETECTOR_CREATE_ERROR != 0 {
637                    error_details.push("Detector creation failed");
638                }
639                if status & STATUS_CONFIG_APPLY_ERROR != 0 {
640                    error_details.push("Configuration apply failed");
641                }
642
643                return Err(RadarError::DeviceError {
644                    message: format!(
645                        "Calibration failed: {} (status: 0x{:08X})",
646                        error_details.join(", "),
647                        status
648                    ),
649                });
650            }
651
652            // Check if calibration completed successfully
653            if (status & STATUS_SENSOR_CALIBRATE_OK) != 0
654                && (status & STATUS_DETECTOR_CREATE_OK) != 0
655            {
656                self.is_calibrated = true;
657                self.last_calibration = Some(Instant::now());
658                info!(
659                    "XM125 calibration completed successfully (status: 0x{:08X})",
660                    status
661                );
662                return Ok(());
663            }
664
665            // Check progress - log which steps have completed
666            let mut progress = Vec::new();
667            if status & STATUS_RSS_REGISTER_OK != 0 {
668                progress.push("RSS");
669            }
670            if status & STATUS_CONFIG_CREATE_OK != 0 {
671                progress.push("Config");
672            }
673            if status & STATUS_SENSOR_CREATE_OK != 0 {
674                progress.push("Sensor");
675            }
676            if status & STATUS_DETECTOR_CREATE_OK != 0 {
677                progress.push("Detector");
678            }
679            if status & STATUS_CONFIG_APPLY_OK != 0 {
680                progress.push("Applied");
681            }
682
683            debug!(
684                "Calibration progress: {} (status: 0x{:08X})",
685                progress.join(", "),
686                status
687            );
688
689            if start_time.elapsed() > CALIBRATION_TIMEOUT {
690                return Err(RadarError::Timeout { timeout: 2 });
691            }
692        }
693    }

822    async fn read_distance_result(&mut self) -> Result<DistanceMeasurement> {
823        // Loop to handle calibration if needed
824        loop {
825            // Read distance result register (4 bytes packed format)
826            let result_data = self.i2c.read_register(REG_DISTANCE_RESULT, 4)?;
827            let result_value = u32::from_be_bytes([
828                result_data[0],
829                result_data[1],
830                result_data[2],
831                result_data[3],
832            ]);
833
834            // Parse packed result format according to distance_reg_protocol.h
835            let num_distances = result_value & 0x0000000F; // Bits 0-3
836            let near_start_edge = (result_value & 0x00000100) != 0; // Bit 8
837            let calibration_needed = (result_value & 0x00000200) != 0; // Bit 9
838            let measure_error = (result_value & 0x00000400) != 0; // Bit 10
839            let temperature_raw = (result_value & 0xFFFF0000) >> 16; // Bits 16-31
840
841            debug!(
842                "Distance result: num_distances={}, near_edge={}, cal_needed={}, error={}",
843                num_distances, near_start_edge, calibration_needed, measure_error
844            );
845
846            if measure_error {
847                debug!("Distance measurement error flag set - may indicate no objects detected");
848                // Note: This error flag can be set when no objects are detected, which is normal
849                // We'll continue processing but log it for debugging
850            }
851
852            if calibration_needed {
853                warn!("Distance detector indicates calibration needed - triggering recalibration");
854                // Trigger recalibration when needed
855                self.calibrate().await?;
856                // Continue loop to re-read result after calibration
857                continue;
858            }
859
860            // Read peak 0 distance and strength (primary measurement)
861            let peak0_distance_data = self.i2c.read_register(REG_PEAK0_DISTANCE, 4)?;
862            let peak0_distance_mm = u32::from_be_bytes([
863                peak0_distance_data[0],
864                peak0_distance_data[1],
865                peak0_distance_data[2],
866                peak0_distance_data[3],
867            ]);
868
869            let peak0_strength_data = self.i2c.read_register(REG_PEAK0_STRENGTH, 4)?;
870            let peak0_strength_raw = u32::from_be_bytes([
871                peak0_strength_data[0],
872                peak0_strength_data[1],
873                peak0_strength_data[2],
874                peak0_strength_data[3],
875            ]);
876
877            // Convert to proper units (values are factor 1000 larger than RSS values)
878            #[allow(clippy::cast_precision_loss)]
879            // Converting mm to meters, precision loss acceptable
880            let distance = if num_distances > 0 {
881                (peak0_distance_mm as f32) / 1000.0 // Convert mm to meters
882            } else {
883                0.0 // No distance detected
884            };
885
886            #[allow(clippy::cast_precision_loss)] // Converting strength, precision loss acceptable
887            let strength = (peak0_strength_raw as f32) / 1000.0; // Convert to proper strength units
888
889            // Convert temperature (signed 16-bit value)
890            #[allow(clippy::cast_possible_wrap)] // Temperature conversion
891            let temperature = temperature_raw as i16;
892
893            debug!(
894                "Parsed distance: {:.2}m, strength: {:.1}, temp: {}°C, peaks: {}",
895                distance, strength, temperature, num_distances
896            );
897
898            self.last_measurement = Some(Instant::now());
899
900            return Ok(DistanceMeasurement {
901                distance,
902                strength,
903                temperature,
904                timestamp: chrono::Utc::now(),
905            });
906        }
907    }

909    /// Measure breathing patterns
910    pub async fn measure_breathing(&mut self) -> Result<BreathingMeasurement> {
911        // Breathing detector needs much longer timeout:
912        // - 5 seconds for distance determination
913        // - 20 seconds for breathing rate estimation
914        // Total: 25+ seconds minimum
915        const BREATHING_TIMEOUT: Duration = Duration::from_secs(30);
916
917        // Auto-connect if not connected and auto-reconnect is enabled
918        if !self.is_connected && self.config.auto_reconnect {
919            info!("Auto-connecting for breathing measurement...");
920            self.auto_connect().await?;
921        }
922
923        // Start breathing application (configuration was applied during connection)
924        info!("Starting breathing analysis - this may take 25+ seconds for full analysis...");
925        self.send_command(CMD_BREATHING_START_APP)?;
926
927        // Wait for breathing analysis to progress through states
928        let start_time = Instant::now();
929        let mut last_state_check = Instant::now();
930
931        while start_time.elapsed() < BREATHING_TIMEOUT {
932            let status = self.get_status_raw()?;
933
934            // Check app state periodically to show progress
935            if last_state_check.elapsed() > Duration::from_secs(2) {
936                let state_data = self.i2c.read_register(REG_BREATHING_APP_STATE, 4)?;
937                let state_raw = u32::from_be_bytes([
938                    state_data[0],
939                    state_data[1],
940                    state_data[2],
941                    state_data[3],
942                ]);
943                let app_state = BreathingAppState::from_u32(state_raw);
944
945                info!(
946                    "Breathing detector state: {:?} ({}s elapsed)",
947                    app_state,
948                    start_time.elapsed().as_secs()
949                );
950                last_state_check = Instant::now();
951
952                // If we reach EstimateBreathingRate state, we can try reading results
953                if matches!(app_state, BreathingAppState::EstimateBreathingRate) {
954                    info!("Breathing rate estimation phase reached - checking for results...");
955                }
956            }
957
958            // Check if not busy (but don't break immediately - breathing detector may cycle busy states)
959            if (status & STATUS_BUSY) == 0 {
960                // Give it a bit more time even when not busy, as breathing analysis is complex
961                tokio::time::sleep(Duration::from_millis(500)).await;
962            } else {
963                tokio::time::sleep(Duration::from_millis(100)).await;
964            }
965        }
966
967        // Read breathing result register (16/0x10)
968        let result_data = self.i2c.read_register(REG_BREATHING_RESULT, 4)?;
969        let result_value = u32::from_be_bytes([
970            result_data[0],
971            result_data[1],
972            result_data[2],
973            result_data[3],
974        ]);
975
976        // Parse breathing result format
977        let result_ready = (result_value & 0x00000001) != 0; // Bit 0: RESULT_READY
978        let result_ready_sticky = (result_value & 0x00000002) != 0; // Bit 1: RESULT_READY_STICKY
979        let temperature_raw = (result_value & 0xFFFF0000) >> 16; // Bits 16-31: TEMPERATURE
980
981        debug!(
982            "Breathing result: ready={}, sticky={}, temp_raw=0x{:04X}",
983            result_ready, result_ready_sticky, temperature_raw
984        );
985
986        // Read breathing rate register (17/0x11)
987        let rate_data = self.i2c.read_register(REG_BREATHING_RATE, 4)?;
988        let rate_raw = u32::from_be_bytes([rate_data[0], rate_data[1], rate_data[2], rate_data[3]]);
989
990        // Read app state register (18/0x12)
991        let state_data = self.i2c.read_register(REG_BREATHING_APP_STATE, 4)?;
992        let state_raw =
993            u32::from_be_bytes([state_data[0], state_data[1], state_data[2], state_data[3]]);
994
995        // Convert values to proper units
996        #[allow(clippy::cast_precision_loss)]
997        // Converting breathing rate, precision loss acceptable
998        let breathing_rate = (rate_raw as f32) / 1000.0; // Factor 1000 larger than RSS value
999        let app_state = BreathingAppState::from_u32(state_raw);
1000
1001        // Convert temperature (signed 16-bit value)
1002        #[allow(clippy::cast_possible_wrap)] // Temperature conversion
1003        let temperature = temperature_raw as i16;
1004
1005        debug!(
1006            "Parsed breathing: rate={:.2} BPM, state={:?}, temp={}°C",
1007            breathing_rate, app_state, temperature
1008        );
1009
1010        self.last_measurement = Some(Instant::now());
1011
1012        Ok(BreathingMeasurement {
1013            result_ready,
1014            breathing_rate,
1015            app_state,
1016            temperature,
1017            timestamp: chrono::Utc::now(),
1018        })
1019    }

1094    /// Automatically connect with retry logic
1095    pub async fn auto_connect(&mut self) -> Result<()> {
1096        const MAX_RETRIES: u32 = 3;
1097        const RETRY_DELAY: Duration = Duration::from_millis(500);
1098
1099        for attempt in 1..=MAX_RETRIES {
1100            match self.connect_async().await {
1101                Ok(()) => {
1102                    info!("Successfully connected to XM125 on attempt {attempt}");
1103                    return Ok(());
1104                }
1105                Err(RadarError::NotConnected) if attempt == 1 => {
1106                    // First connection failure - try resetting XM125 to run mode
1107                    warn!(
1108                        "XM125 not detected on I2C bus, attempting hardware reset to run mode..."
1109                    );
1110                    if let Err(reset_err) = self.reset_xm125_to_run_mode() {
1111                        warn!("Failed to reset XM125: {reset_err}");
1112                    } else {
1113                        info!("XM125 reset completed, retrying connection...");
1114                        // Give the module time to initialize after reset
1115                        tokio::time::sleep(Duration::from_millis(1000)).await;
1116                    }
1117                }
1118                Err(e) => {
1119                    if attempt < MAX_RETRIES {
1120                        warn!("Connection attempt {attempt} failed: {e}. Retrying...");
1121                        tokio::time::sleep(RETRY_DELAY).await;
1122                    } else {
1123                        return Err(e);
1124                    }
1125                }
1126            }
1127        }
1128        Err(RadarError::NotConnected)
1129    }

1705        let start_time = Instant::now();
1706
1707        loop {
1708            let status = self.get_status_raw()?;
1709

1711                return Ok(());
1712            }
1713
1714            if status & STATUS_ERROR != 0 {
1715                return Err(RadarError::DeviceError {
1716                    message: "Device error during measurement".to_string(),
1717                });
1718            }
1719
1720            if start_time.elapsed() > MEASUREMENT_TIMEOUT {
1721                return Err(RadarError::Timeout { timeout: 5 });
1722            }
1723
1724            tokio::time::sleep(Duration::from_millis(10)).await;
1725        }
1726    }
