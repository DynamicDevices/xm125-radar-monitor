216/// Configure distance measurement range
217fn configure_distance_range(radar: &mut XM125Radar, range_str: &str) -> Result<(), RadarError> {
218    let parts: Vec<&str> = range_str.split(':').collect();
219    if parts.len() != 2 {
220        return Err(RadarError::DeviceError {
221            message: format!(
222                "Invalid range format '{range_str}'. Use 'start:end' (e.g., '0.1:3.0')"
223            ),
224        });
225    }
226
227    let start: f32 = parts[0].parse().map_err(|_| RadarError::DeviceError {
228        message: format!("Invalid start range: {}", parts[0]),
229    })?;
230
231    let end: f32 = parts[1].parse().map_err(|_| RadarError::DeviceError {
232        message: format!("Invalid end range: {}", parts[1]),
233    })?;
234
235    if start >= end {
236        return Err(RadarError::DeviceError {
237            message: format!("Start range ({start}) must be less than end range ({end})"),
238        });
239    }
240
241    if start < 0.1 || end > 3.0 {
242        return Err(RadarError::DeviceError {
243            message: "Distance range must be between 0.1m and 3.0m".to_string(),
244        });
245    }
246
247    info!("ğŸ¯ Configuring distance range: {start:.2}m - {end:.2}m");
248    radar.config.start_m = start;
249    radar.config.length_m = end - start;
250
251    Ok(())
252}

254/// Configure presence parameters for the radar
255fn configure_presence_parameters(
256    radar: &mut radar::XM125Radar,
257    presence_range: Option<&PresenceRange>,
258    min_range: Option<f32>,
259    max_range: Option<f32>,
260    sensitivity: Option<f32>,
261    frame_rate: Option<f32>,
262) -> Result<(), RadarError> {
263    let mut config_changed = false;
264
265    // Configure range (either preset or custom)
266    if let Some(range) = presence_range {
267        info!("ğŸ¯ Configuring presence range preset: {range:?}");
268        radar.config.presence_range = range.clone().into();
269        config_changed = true;
270    } else if let (Some(min), Some(max)) = (min_range, max_range) {
271        info!("ğŸ¯ Configuring custom presence range: {min:.2}m - {max:.2}m");
272
273        // Validate range
274        if min >= max {
275            return Err(RadarError::DeviceError {
276                message: format!(
277                    "Minimum range ({min:.2}m) must be less than maximum range ({max:.2}m)"
278                ),
279            });
280        }
281
282        if min < 0.06 || max > 7.0 {
283            return Err(RadarError::DeviceError {
284                message: "Presence range must be between 0.06m and 7.0m".to_string(),
285            });
286        }
287
288        // Set custom range (this will be used by configure_presence_range)
289        radar.config.start_m = min;
290        radar.config.length_m = max - min;
291        config_changed = true;
292    }
293
294    // Configure sensitivity
295    if let Some(sens) = sensitivity {
296        info!("ğŸ¯ Configuring sensitivity: {sens:.2}");
297
298        if !(0.1..=5.0).contains(&sens) {
299            return Err(RadarError::DeviceError {
300                message: format!("Sensitivity must be between 0.1 and 5.0 (got {sens:.2})"),
301            });
302        }
303
304        // Convert sensitivity to internal threshold values
305        radar.config.intra_detection_threshold = sens * 1000.0; // Convert to internal units
306        radar.config.inter_detection_threshold = sens * 800.0; // Slightly lower for inter
307        config_changed = true;
308    }
309
310    // Configure frame rate
311    if let Some(rate) = frame_rate {
312        info!("ğŸ¯ Configuring frame rate: {rate:.1} Hz");
313
314        if !(1.0..=60.0).contains(&rate) {
315            return Err(RadarError::DeviceError {
316                message: format!("Frame rate must be between 1.0 and 60.0 Hz (got {rate:.1})"),
317            });
318        }
319
320        radar.config.frame_rate = rate;
321        config_changed = true;
322    }
323
324    // Apply configuration to hardware if anything changed OR if no range was specified
325    // (to ensure default long range is properly applied)
326    if config_changed || (presence_range.is_none() && min_range.is_none() && max_range.is_none()) {
327        radar.configure_presence_range();
328        if config_changed {
329            info!("âœ… Presence parameters configured successfully");
330        } else {
331            info!("âœ… Applied default presence configuration (long range: 0.5m - 7.0m)");
332        }
333    }
334
335    Ok(())
336}

357            Err(e) => {

359                warn!("Failed to debug registers: {e}");

361        }

363        eprintln!("âŒ Cannot debug registers: radar not connected after connection attempt");
364    }

674                        }
675                    } else {
676                        "NONE"
677                    };
678
679                    writer
680                        .write_record([
681                            timestamp_full.as_str(),
682                            &result.presence_detected.to_string(),
683                            &result.presence_distance.to_string(),
684                            &result.intra_presence_score.to_string(),
685                            &result.inter_presence_score.to_string(),
686                            intra_strength,
687                            inter_strength,
688                            confidence,
689                            &measurement_count.to_string(),
690                        ])
691                        .map_err(|e| RadarError::DeviceError {
692                            message: format!("Failed to write CSV record: {e}"),
693                        })?;
694                    writer.flush().map_err(|e| RadarError::DeviceError {
695                        message: format!("Failed to flush CSV writer: {e}"),
696                    })?;
697                }
698
699                // Check if we've reached the target count
700                if !infinite && measurement_count >= total_count {
701                    break;
702                }
703            }
704            Err(e) => {
705                eprintln!("âŒ Measurement #{} failed: {}", measurement_count + 1, e);
706                // Continue with next measurement
707            }
708        }
709
710        // Wait for the specified interval
711        let elapsed = start_time.elapsed();
712        let target_duration = Duration::from_millis(interval);
713        if elapsed < target_duration {
714            sleep(target_duration - elapsed).await;
715        }
716    }
717
718    if !cli.quiet {
719        println!("â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€");
720        println!("âœ… Completed {measurement_count} presence measurements");
721        if let Some(filename) = save_to {
722            println!("ğŸ’¾ Results saved to: {filename}");
723        }
724    }
725
726    Ok(())
727}

733    firmware_path: &str,

740        }
741        FirmwareAction::Update {

744            verify: _,
745        } => {
746            let manager =
747                firmware::FirmwareManager::new(firmware_path, "/usr/bin/xm125-control.sh", 0x52);
748            manager.update_firmware(*firmware_type).await?;
749        }
750        FirmwareAction::Verify { firmware_type } => {
751            info!("Firmware verification not yet implemented in v2.0.0");

753                info!("Would verify firmware type: {fw_type:?}");
754            } else {
755                info!("Would verify current firmware");
756            }
757        }
758        FirmwareAction::Erase { .. }
759        | FirmwareAction::Checksum { .. }
760        | FirmwareAction::Bootloader { .. } => {
761            // These are handled earlier in run() before I2C initialization
762            unreachable!("These firmware commands should be handled before I2C initialization");
763        }
764    }
765    Ok(())
766}

794/// Handle firmware checksum command
795fn handle_firmware_checksum_command(
796    firmware_type: Option<&firmware::FirmwareType>,
797    verbose: bool,
798    firmware_path: &str,
799) -> Result<(), RadarError> {
800    let manager = firmware::FirmwareManager::new(firmware_path, "/usr/bin/xm125-control.sh", 0x52);
801
802    if let Some(fw_type) = firmware_type {
803        let checksum = manager.calculate_binary_checksum(*fw_type)?;
804        if verbose {
805            println!(
806                "Firmware: {} ({})",
807                fw_type.display_name(),
808                fw_type.binary_filename()
809            );
810            println!("MD5: {checksum}");
811        } else {
812            println!("{}: {}", fw_type.display_name(), checksum);
813        }
814    } else {
815        // Calculate checksums for all firmware types
816        for fw_type in [
817            firmware::FirmwareType::Distance,
818            firmware::FirmwareType::Presence,
819        ] {
820            match manager.calculate_binary_checksum(fw_type) {
821                Ok(checksum) => {
822                    if verbose {
823                        println!(
824                            "Firmware: {} ({})",
825                            fw_type.display_name(),
826                            fw_type.binary_filename()
827                        );
828                        println!("MD5: {checksum}");
829                        println!();
830                    } else {
831                        println!("{}: {}", fw_type.display_name(), checksum);
832                    }
833                }
834                Err(e) => {
835                    eprintln!(
836                        "Error calculating checksum for {}: {}",
837                        fw_type.display_name(),
838                        e
839                    );
840                }
841            }
842        }
843    }
844    Ok(())
845}
