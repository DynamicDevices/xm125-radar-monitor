10/// Configure distance measurement range
11pub fn configure_distance_range(radar: &mut XM125Radar, range_str: &str) -> Result<(), RadarError> {
12    let parts: Vec<&str> = range_str.split(':').collect();
13    if parts.len() != 2 {
14        return Err(RadarError::DeviceError {
15            message: format!(
16                "Invalid range format '{range_str}'. Use 'start:end' (e.g., '0.1:3.0')"
17            ),
18        });
19    }
20
21    let start: f32 = parts[0].parse().map_err(|_| RadarError::DeviceError {
22        message: format!("Invalid start range: {}", parts[0]),
23    })?;
24
25    let end: f32 = parts[1].parse().map_err(|_| RadarError::DeviceError {
26        message: format!("Invalid end range: {}", parts[1]),
27    })?;
28
29    if start >= end {
30        return Err(RadarError::DeviceError {
31            message: format!("Start range ({start}) must be less than end range ({end})"),
32        });
33    }
34
35    info!("üéØ Configuring distance range: {start:.2}m - {end:.2}m");
36    radar.config.start_m = start;
37    radar.config.length_m = end - start;
38    Ok(())
39}

41/// Configure presence parameters for the radar
42#[allow(unused_assignments)]
43pub fn configure_presence_parameters(
44    radar: &mut XM125Radar,
45    presence_range: Option<&PresenceRange>,
46    min_range: Option<f32>,
47    max_range: Option<f32>,
48    sensitivity: Option<f32>,
49    frame_rate: Option<f32>,
50    profile: &ProfileMode,
51) -> Result<(), RadarError> {
52    #[allow(unused_assignments)]
53    let mut config_changed = false;
54
55    // Configure range (either preset or custom)
56    if let Some(range) = presence_range {
57        radar.config.presence_range = match range {
58            PresenceRange::Short => RadarPresenceRange::Short,
59            PresenceRange::Medium => RadarPresenceRange::Medium,
60            PresenceRange::Long => RadarPresenceRange::Long,
61        };
62        config_changed = true;
63        info!("üéØ Set presence range: {range:?}");
64    }
65
66    // Custom range overrides preset
67    if let (Some(min), Some(max)) = (min_range, max_range) {
68        if min >= max {
69            return Err(RadarError::DeviceError {
70                message: format!("min_range ({min}) must be less than max_range ({max})"),
71            });
72        }
73        radar.config.start_m = min;
74        radar.config.length_m = max - min;
75        config_changed = true;
76        info!("üéØ Set custom presence range: {min:.2}m - {max:.2}m");
77    }
78
79    // Configure sensitivity
80    if let Some(sens) = sensitivity {
81        if !(0.1..=5.0).contains(&sens) {
82            return Err(RadarError::DeviceError {
83                message: format!("Sensitivity must be between 0.1 and 5.0 (got {sens:.2})"),
84            });
85        }
86        // Convert sensitivity to internal threshold values
87        radar.config.intra_detection_threshold = sens * 1000.0;
88        radar.config.inter_detection_threshold = sens * 800.0;
89        config_changed = true;
90        info!("üîß Set sensitivity: {sens:.2}");
91    }
92
93    // Configure frame rate
94    if let Some(rate) = frame_rate {
95        if !(1.0..=60.0).contains(&rate) {
96            return Err(RadarError::DeviceError {
97                message: format!("Frame rate must be between 1.0 and 60.0 Hz (got {rate:.1})"),
98            });
99        }
100        radar.config.frame_rate = rate;
101        config_changed = true;
102        info!("üîß Set frame rate: {rate:.1} Hz");
103    }
104
105    // Configure profile mode
106    match profile {
107        ProfileMode::Auto => {
108            radar.config.auto_profile_enabled = true;
109            info!("üîß Using automatic profile selection (recommended)");
110        }
111        ProfileMode::Manual => {
112            radar.config.auto_profile_enabled = false;
113            info!("üîß Using manual profile selection (Profile 5 for 7m range)");
114        }
115    }
116    config_changed = true;
117
118    if config_changed || (presence_range.is_none() && min_range.is_none() && max_range.is_none()) {
119        radar.configure_presence_range()?;
120        if config_changed {
121            info!("‚úÖ Presence parameters configured successfully");
122        } else {
123            info!("‚úÖ Applied default presence configuration (long range: 0.5m - 7.0m)");
124        }
125    }
126    Ok(())
127}

129/// Debug registers if radar is connected, with automatic connection attempt
130pub fn debug_registers_if_connected(radar: &mut XM125Radar, mode: &str) {
131    if radar.is_connected() {
132        match radar.debug_registers(mode) {
133            Ok(_) => info!("‚úÖ Register debugging completed successfully"),
134            Err(e) => {
135                eprintln!("‚ùå Failed to debug registers: {e}");
136                warn!("Register debugging failed, but continuing with measurement");
137            }
138        }
139    } else {
140        warn!("‚ö†Ô∏è  Radar not connected, skipping register debugging");
141        warn!("   Use --verbose for detailed connection diagnostics");
142    }
