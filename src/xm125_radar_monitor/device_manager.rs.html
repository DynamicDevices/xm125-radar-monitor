49    /// Check device presence on I2C bus (non-intrusive)
50    pub async fn check_device_presence(&self) -> DeviceState {
51        info!("üîç Checking XM125 device presence...");
52
53        // First check if device appears on I2C bus using i2cdetect
54        let i2c_bus = self.extract_i2c_bus_number();
55        let present_on_bus = self.check_i2c_bus_presence(i2c_bus).await;
56
57        if !present_on_bus {
58            info!("‚ùå XM125 not detected on I2C bus {}", i2c_bus);
59            return DeviceState {
60                is_present: false,
61                is_responsive: false,
62                firmware_type: None,
63                app_id: None,
64                needs_reset: true,
65            };
66        }
67
68        info!(
69            "‚úÖ XM125 detected on I2C bus {} at address 0x{:02X}",
70            i2c_bus, self.i2c_address
71        );
72
73        // Try to communicate with device
74        match self.check_device_responsiveness().await {
75            Ok((app_id, firmware_type)) => {
76                info!(
77                    "‚úÖ XM125 is responsive - {} (App ID: {})",
78                    firmware_type.display_name(),
79                    app_id
80                );
81                DeviceState {
82                    is_present: true,
83                    is_responsive: true,
84                    firmware_type: Some(firmware_type),
85                    app_id: Some(app_id),
86                    needs_reset: false,
87                }
88            }
89            Err(e) => {
90                warn!("‚ö†Ô∏è  XM125 present but not responsive: {}", e);
91                DeviceState {
92                    is_present: true,
93                    is_responsive: false,
94                    firmware_type: None,
95                    app_id: None,
96                    needs_reset: true,
97                }
98            }
99        }
100    }

102    /// Check if device is responsive and read firmware info
103    async fn check_device_responsiveness(&self) -> Result<(u32, FirmwareType)> {
104        let i2c_device = I2cDevice::new(&self.i2c_device_path, self.i2c_address)?;
105        let mut radar = XM125Radar::new(i2c_device);
106
107        // Try to connect (with timeout)
108        let connect_result =
109            tokio::time::timeout(Duration::from_secs(3), async { radar.connect() }).await;
110
111        match connect_result {
112            Ok(Ok(())) => {
113                // Successfully connected, read application ID
114                let app_id = radar.read_application_id()?;
115                let firmware_type = FirmwareType::from_app_id(app_id);
116                Ok((app_id, firmware_type))
117            }
118            Ok(Err(e)) => Err(e),
119            Err(_) => Err(RadarError::Timeout { timeout: 3 }),
120        }
121    }

123    /// Reset device to run mode
124    pub async fn reset_to_run_mode(&self) -> Result<()> {
125        info!("üîÑ Resetting XM125 to run mode...");
126
127        self.firmware_manager.reset_to_run_mode().await?;
128
129        // Wait for device to initialize
130        sleep(Duration::from_millis(1000)).await;
131
132        // Verify device is now responsive
133        let state = self.check_device_presence().await;
134        if state.is_responsive {
135            info!("‚úÖ XM125 successfully reset to run mode");
136            Ok(())
137        } else {
138            Err(RadarError::DeviceError {
139                message: "Device reset completed but device is still not responsive".to_string(),
140            })
141        }
142    }

144    /// Update firmware to specified type with verification
145    pub async fn update_firmware(&self, target_type: FirmwareType, verify: bool) -> Result<()> {
146        info!(
147            "üöÄ Updating firmware to {} (verify: {})",
148            target_type.display_name(),
149            verify
150        );
151
152        // Check current state
153        let state = self.check_device_presence().await;
154
155        // If device needs reset, do it first
156        if state.needs_reset {
157            self.reset_to_run_mode().await?;
158        }
159
160        // Check if update is actually needed
161        if let Some(current_type) = state.firmware_type {
162            if current_type == target_type {
163                info!(
164                    "‚úÖ Firmware already matches target type: {}",
165                    target_type.display_name()
166                );
167                return Ok(());
168            }
169        }
170
171        // Perform firmware update
172        if verify {
173            self.firmware_manager
174                .update_firmware_with_verification(target_type, true)
175                .await?;
176        } else {
177            self.firmware_manager.update_firmware(target_type).await?;
178        }
179
180        // Verify the update worked
181        sleep(Duration::from_millis(1000)).await;
182        let final_state = self.check_device_presence().await;
183
184        match final_state.firmware_type {
185            Some(actual_type) if actual_type == target_type => {
186                info!(
187                    "‚úÖ Firmware update successful: {}",
188                    target_type.display_name()
189                );
190                Ok(())
191            }
192            Some(actual_type) => Err(RadarError::DeviceError {
193                message: format!(
194                    "Firmware update failed: Expected {}, got {}",
195                    target_type.display_name(),
196                    actual_type.display_name()
197                ),
198            }),
199            None => Err(RadarError::DeviceError {
200                message: "Firmware update failed: Device not responsive after update".to_string(),
201            }),
202        }
203    }

205    /// Get comprehensive device information
206    #[allow(dead_code)] // Reserved for future use
207    pub async fn get_device_info(&self) -> Result<String> {
208        let state = self.check_device_presence().await;
209
210        let mut info = String::new();
211        info.push_str(&format!(
212            "Device Present: {}\n",
213            if state.is_present { "YES" } else { "NO" }
214        ));
215        info.push_str(&format!(
216            "Device Responsive: {}\n",
217            if state.is_responsive { "YES" } else { "NO" }
218        ));
219
220        if let Some(firmware_type) = state.firmware_type {
221            info.push_str(&format!(
222                "Firmware Type: {}\n",
223                firmware_type.display_name()
224            ));
225        }
226
227        if let Some(app_id) = state.app_id {
228            info.push_str(&format!("Application ID: {}\n", app_id));
229        }
230
231        info.push_str(&format!(
232            "Needs Reset: {}",
233            if state.needs_reset { "YES" } else { "NO" }
234        ));
235
236        Ok(info)
237    }

239    /// Extract I2C bus number from device path
240    fn extract_i2c_bus_number(&self) -> u8 {
241        // Extract bus number from path like "/dev/i2c-2"
242        self.i2c_device_path
243            .split('-')
244            .next_back()
245            .and_then(|s| s.parse().ok())
246            .unwrap_or(2) // Default to bus 2 for Sentai
247    }

249    /// Check if device is present on I2C bus using i2cdetect
250    async fn check_i2c_bus_presence(&self, bus: u8) -> bool {
251        use std::process::Command;
252
253        let output = Command::new("i2cdetect")
254            .args(["-y", &bus.to_string()])
255            .output();
256
257        match output {
258            Ok(output) if output.status.success() => {
259                let stdout = String::from_utf8_lossy(&output.stdout);
260                let address_str = format!("{:02x}", self.i2c_address);
261                stdout.contains(&address_str)
262            }
263            _ => false,
264        }
265    }
