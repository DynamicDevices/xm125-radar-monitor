31    /// Configure distance range
32    pub fn configure_range(&mut self, start_m: f32, length_m: f32) -> Result<()> {
33        let start_mm = (start_m * 1000.0) as u32;
34        let end_mm = ((start_m + length_m) * 1000.0) as u32;
35        
36        info!("Configuring distance range: {:.3}m to {:.3}m", start_m, start_m + length_m);
37        
38        // Write range configuration to registers
39        self.i2c.write_register(REG_START_CONFIG, &start_mm.to_be_bytes())?;
40        self.i2c.write_register(REG_END_CONFIG, &end_mm.to_be_bytes())?;
41        
42        info!("âœ… Distance range configured");
43        Ok(())
44    }

67    /// Check if distance detector is busy
68    pub fn is_busy(&mut self) -> Result<bool> {
69        let status = self.i2c.read_register(REG_DETECTOR_STATUS, 4)?;
70        let status_value = u32::from_be_bytes([status[0], status[1], status[2], status[3]]);
71        Ok((status_value & STATUS_BUSY_MASK) != 0)
72    }
73
74    /// Check if distance detector has errors
75    pub fn has_errors(&mut self) -> Result<bool> {
76        let status = self.i2c.read_register(REG_DETECTOR_STATUS, 4)?;
77        let status_value = u32::from_be_bytes([status[0], status[1], status[2], status[3]]);
78        Ok((status_value & STATUS_ERROR_MASK) != 0)
79    }

81    /// Wait for distance detector to not be busy
82    pub async fn wait_for_not_busy(&mut self, timeout: Duration) -> Result<()> {
83        let start = std::time::Instant::now();
84        while start.elapsed() < timeout {
85            if !self.is_busy()? {
86                return Ok(());
87            }
88            tokio::time::sleep(Duration::from_millis(10)).await;
89        }
90        Err(RadarError::Timeout { timeout: timeout.as_secs() })
91    }
92
93    /// Write command safely with busy/error checking
94    pub async fn write_command_safe(&mut self, command: u32) -> Result<()> {
95        // Check if detector is busy before writing command
96        if self.is_busy()? {
97            self.wait_for_not_busy(Duration::from_secs(5)).await?;
98        }
99
100        // Check for errors - if present, only RESET MODULE command is allowed
101        if self.has_errors()? && command != CMD_RESET_MODULE {
102            warn!("Distance detector has errors, resetting module before command");
103            self.reset_module().await?;
104        }
105
106        // Write the command
107        self.i2c.write_register(REG_COMMAND, &command.to_be_bytes())?;
108        Ok(())
109    }

111    /// Reset the distance module
112    pub async fn reset_module(&mut self) -> Result<()> {
113        info!("ðŸ”„ Resetting XM125 distance module...");
114        
115        // RESET MODULE command can always be sent, even when there are errors
116        self.i2c.write_register(REG_COMMAND, &CMD_RESET_MODULE.to_be_bytes())?;
117        
118        // Wait for reset to complete
119        tokio::time::sleep(Duration::from_millis(500)).await;
120        
121        info!("âœ… XM125 distance module reset completed");
122        Ok(())
123    }

125    /// Apply configuration and calibrate
126    pub async fn apply_config_and_calibrate(&mut self) -> Result<()> {
127        info!("Applying distance detector configuration and calibrating...");
128        self.write_command_safe(CMD_APPLY_CONFIG_AND_CALIBRATE).await?;
129        
130        // Wait for configuration and calibration to complete
131        self.wait_for_not_busy(CALIBRATION_TIMEOUT).await?;
132        
133        // Check for configuration errors
134        if self.has_errors()? {
135            return Err(RadarError::DeviceError {
136                message: "Distance detector configuration/calibration failed".to_string(),
137            });
138        }
139        
140        info!("âœ… Distance detector configured and calibrated successfully");
141        Ok(())
142    }

144    /// Measure distance
145    pub async fn measure(&mut self) -> Result<DistanceMeasurement> {
146        // Send measure command
147        self.write_command_safe(CMD_MEASURE_DISTANCE).await?;
148        
149        // Wait for measurement to complete
150        self.wait_for_not_busy(MEASUREMENT_TIMEOUT).await?;
151        
152        // Read measurement results
153        let distance_result = self.i2c.read_register(REG_DISTANCE_RESULT, 4)?;
154        let strength_result = self.i2c.read_register(REG_PEAK0_STRENGTH, 4)?;
155        
156        // Parse results
157        let distance_value = u32::from_be_bytes([distance_result[0], distance_result[1], distance_result[2], distance_result[3]]);
158        let strength_value = u32::from_be_bytes([strength_result[0], strength_result[1], strength_result[2], strength_result[3]]);
159        
160        // Convert distance from mm to meters
161        let distance = (distance_value as f32) / 1000.0;
162        
163        // Convert strength (scaled appropriately)
164        let strength = strength_value as f32;
165        
166        // Mock temperature for now (would need additional register read)
167        let temperature = 25i16;
168
169        Ok(DistanceMeasurement {
170            distance,
171            strength,
172            temperature,
173            timestamp: chrono::Utc::now(),
174        })
175    }
