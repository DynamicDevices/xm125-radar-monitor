38    /// Calculate optimal profile based on detection range
39    fn calculate_optimal_profile(_start_mm: u32, end_mm: u32) -> u32 {
40        let max_range_m = end_mm as f32 / 1000.0;
41        
42        match max_range_m {
43            r if r <= 0.7 => 1,   // Profile 1: Short range (0.06m - 0.7m)
44            r if r <= 2.0 => 2,   // Profile 2: Short-medium range (0.2m - 2.0m)  
45            r if r <= 3.5 => 3,   // Profile 3: Medium range (0.3m - 3.5m)
46            r if r <= 6.0 => 4,   // Profile 4: Medium-long range (0.5m - 6.0m)
47            _ => 5,               // Profile 5: Long range (1.0m - 7.0m+)
48        }
49    }

51    /// Calculate optimal step length based on maximum range
52    fn calculate_optimal_step_length(end_mm: u32) -> u32 {
53        let max_range_m = end_mm as f32 / 1000.0;
54        
55        // Step length should be approximately range_length / 60 for optimal performance
56        // But with minimum and maximum bounds
57        let calculated = (max_range_m * 1000.0 / 60.0) as u32;
58        
59        // Clamp to reasonable bounds
60        calculated.clamp(12, 120)
61    }

63    /// Configure presence range according to datasheet specifications
64    pub fn configure_range(&mut self, range: PresenceRange, custom_start_m: Option<f32>, custom_length_m: Option<f32>) -> Result<(u32, u32)> {
65        info!("üéØ Configuring presence range preset: {:?}", range);
66        
67        // Convert PresenceRange enum to millimeter values according to datasheet
68        let (start_mm, end_mm) = match range {
69            PresenceRange::Short => {
70                // Short range: 6cm to 70cm (60mm to 700mm)
71                (60u32, 700u32)
72            }
73            PresenceRange::Medium => {
74                // Medium range: 20cm to 2m (200mm to 2000mm)  
75                (200u32, 2000u32)
76            }
77            PresenceRange::Long => {
78                // Long range: 30cm to 5.5m (300mm to 5500mm) - Philip's working config
79                (300u32, 5500u32)
80            }
81        };
82        
83        // Check for custom range override (when min/max range is explicitly set)
84        let (final_start_mm, final_end_mm) = if let (Some(start_m), Some(length_m)) = (custom_start_m, custom_length_m) {
85            // Custom range specified via CLI --min-range and --max-range
86            let start_mm = (start_m * 1000.0) as u32;
87            let end_mm = ((start_m + length_m) * 1000.0) as u32;
88            info!("Using custom range: {:.3}m to {:.3}m", start_m, start_m + length_m);
89            (start_mm, end_mm)
90        } else {
91            // Use preset range
92            info!("Using preset range: {}mm to {}mm ({:.1}m to {:.1}m)", 
93                  start_mm, end_mm, start_mm as f32 / 1000.0, end_mm as f32 / 1000.0);
94            (start_mm, end_mm)
95        };
96        
97        // Calculate optimal profile based on actual range being used
98        let optimal_profile = Self::calculate_optimal_profile(final_start_mm, final_end_mm);
99        let optimal_step_length = Self::calculate_optimal_step_length(final_end_mm);
100        
101        info!("üéØ Selected Profile {} for range {}mm-{}mm ({:.1}m-{:.1}m)", 
102              optimal_profile, final_start_mm, final_end_mm, 
103              final_start_mm as f32 / 1000.0, final_end_mm as f32 / 1000.0);
104        
105        info!("‚úÖ Presence range parameters calculated");
106        Ok((optimal_profile, optimal_step_length))
107    }

150    /// Apply the complete configuration including range settings
151    pub fn apply_complete_configuration(&mut self, final_start_mm: u32, final_end_mm: u32) -> Result<()> {
152        // CRITICAL: Reset module before applying new configuration (from datasheet requirement)
153        info!("Resetting presence detector module before configuration...");
154        self.reset_module()?;
155        
156        // Wait for reset to complete
157        info!("Waiting for module reset to complete...");
158        self.wait_for_not_busy()?;
159        
160        // CRITICAL: Write range values LAST to prevent them being overwritten by profile settings
161        info!("Writing start range to register 0x{:04X} ({}): {} mm", 
162              PRESENCE_REG_START_ADDRESS, PRESENCE_REG_START_ADDRESS, final_start_mm);
163        self.i2c.write_register(PRESENCE_REG_START_ADDRESS, &final_start_mm.to_be_bytes())?;
164        
165        info!("Writing end range to register 0x{:04X} ({}): {} mm", 
166              PRESENCE_REG_END_ADDRESS, PRESENCE_REG_END_ADDRESS, final_end_mm);
167        self.i2c.write_register(PRESENCE_REG_END_ADDRESS, &final_end_mm.to_be_bytes())?;
168        
169        info!("‚úÖ Range configuration written to hardware registers");
170        
171        // CRITICAL: Apply configuration by writing CMD_PRESENCE_APPLY_CONFIGURATION to command register 0x0100
172        // Without this step, detector uses default values (end point = 2500mm)
173        info!("Applying presence detector configuration (CMD_PRESENCE_APPLY_CONFIGURATION to register 0x0100)");
174        self.i2c.write_register(PRESENCE_REG_COMMAND_ADDRESS, &CMD_PRESENCE_APPLY_CONFIGURATION.to_be_bytes())?;
175        
176        // CRITICAL: Wait for the configuration to be done (from example code)
177        info!("Waiting for configuration to complete...");
178        self.wait_for_not_busy()?;
179        
180        // CRITICAL: Test if configuration of detector was OK (from example code)
181        info!("Verifying configuration was applied successfully...");
182        if !self.configuration_ok()? {
183            return Err(RadarError::DeviceError {
184                message: "Configuration verification failed - detector did not accept the configuration".to_string(),
185            });
186        }
187        info!("‚úÖ Configuration verified successfully");
188        
189        // CRITICAL: Start the detector after configuration
190        info!("Starting presence detector (CMD_PRESENCE_START_DETECTOR to register 0x0100)");
191        self.i2c.write_register(PRESENCE_REG_COMMAND_ADDRESS, &CMD_PRESENCE_START_DETECTOR.to_be_bytes())?;
192        
193        info!("‚úÖ Presence detector configured and started - full range should now be available");
194        Ok(())
195    }

197    /// Wait for detector to not be busy (from example code)
198    fn wait_for_not_busy(&mut self) -> Result<()> {
199        let timeout = std::time::Duration::from_secs(5);
200        let start = std::time::Instant::now();
201        
202        while start.elapsed() < timeout {
203            if !self.is_busy()? {
204                return Ok(());
205            }
206            std::thread::sleep(std::time::Duration::from_millis(10));
207        }
208        
209        Err(RadarError::Timeout {
210            timeout: timeout.as_secs(),
211        })
212    }

214    /// Check if configuration was applied successfully (from example code)
215    fn configuration_ok(&mut self) -> Result<bool> {
216        // Read detector status to check for configuration success
217        let status_data = self.i2c.read_register(PRESENCE_REG_DETECTOR_STATUS_ADDRESS, 4)?;
218        let status = u32::from_be_bytes([status_data[0], status_data[1], status_data[2], status_data[3]]);
219        
220        // Check if there are any error bits set (bit 28 and others)
221        let has_errors = (status & 0x10000000) != 0; // Error bit
222        
223        if has_errors {
224            warn!("Configuration failed - detector status shows errors: 0x{:08X}", status);
225            return Ok(false);
226        }
227        
228        Ok(true)
229    }

257    /// Write command safely with busy/error checking (section 2.3.1 compliance)
258    pub async fn write_command_safe(&mut self, command: u32) -> Result<()> {
259        // Check if detector is busy before writing command
260        if self.is_busy()? {
261            self.wait_for_not_busy()?;
262        }
263
264        // Check for errors - if present, only RESET MODULE command is allowed
265        if self.has_errors()? && command != CMD_RESET_MODULE {
266            warn!("Presence detector has errors, resetting module before command");
267            self.reset_module()?;
268        }
269
270        // Write the command
271        self.i2c.write_register(PRESENCE_REG_COMMAND_ADDRESS, &command.to_be_bytes())?;
272        Ok(())
273    }

330    /// Measure presence detection
331    pub async fn measure(&mut self) -> Result<PresenceMeasurement> {
332        // Read presence detection results
333        let presence_result = self.i2c.read_register(REG_PRESENCE_RESULT, 4)?;
334        let presence_distance = self.i2c.read_register(REG_PRESENCE_DISTANCE, 4)?;
335        let intra_score = self.i2c.read_register(REG_INTRA_PRESENCE_SCORE, 4)?;
336        let inter_score = self.i2c.read_register(REG_INTER_PRESENCE_SCORE, 4)?;
337
338        // Parse results
339        let presence_value = u32::from_be_bytes([presence_result[0], presence_result[1], presence_result[2], presence_result[3]]);
340        let distance_value = u32::from_be_bytes([presence_distance[0], presence_distance[1], presence_distance[2], presence_distance[3]]);
341        let intra_value = u32::from_be_bytes([intra_score[0], intra_score[1], intra_score[2], intra_score[3]]);
342        let inter_value = u32::from_be_bytes([inter_score[0], inter_score[1], inter_score[2], inter_score[3]]);
343
344        // Extract presence detection (bit 0)
345        let presence_detected = (presence_value & 0x1) != 0;
346        
347        // Convert distance from mm to meters
348        let presence_distance = (distance_value as f32) / 1000.0;
349        
350        // Convert scores (scaled by 1000)
351        let intra_presence_score = (intra_value as f32) / 1000.0;
352        let inter_presence_score = (inter_value as f32) / 1000.0;
353
354        Ok(PresenceMeasurement {
355            presence_detected,
356            presence_distance,
357            intra_presence_score,
358            inter_presence_score,
359            timestamp: chrono::Utc::now(),
360        })
361    }
