126    /// Debug a single register
127    fn debug_register(&mut self, address: u16, name: &str, description: &str) -> Result<()> {
128        match self.i2c.read_register(address, 4) {
129            Ok(data) => {
130                let value = u32::from_be_bytes([data[0], data[1], data[2], data[3]]);
131                println!("  0x{:04X} ({:3}) â”‚ {:<25} â”‚ 0x{:08X} ({:10}) â”‚ {}", 
132                        address, address, name, value, value, description);
133            }
134            Err(e) => {
135                println!("  0x{:04X} ({:3}) â”‚ {:<25} â”‚ ERROR: {:?} â”‚ {}", 
136                        address, address, name, e, description);
137            }
138        }
139        Ok(())
140    }

144        println!("================================================================================");
145        println!("XM125 Register Dump - {} Mode", detector_mode);
146        println!("================================================================================");

148        self.debug_common_registers()?;
149
150        match detector_mode.to_lowercase().as_str() {
151            "presence" => self.debug_presence_registers()?,
152            "distance" => self.debug_distance_registers()?,
153            _ => {
154                // Debug both for unknown modes
155                self.debug_presence_registers()?;
156                self.debug_distance_registers()?;
157            }
158        }

160        println!("================================================================================");
